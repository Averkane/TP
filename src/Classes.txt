public abstract class Abonne {

    protected String nom;
    protected String tel;
    protected Adresse adresse;

    public Abonne() {}

    public void saisir() {
        Scanner sc = new Scanner(System.in);

        System.out.println("Nom : ");
        nom = sc.nextLine();

        System.out.println("Téléphone : ");
        tel = sc.nextLine();

        System.out.println("Adresse : ");
        adresse = new Adresse();
        adresse.saisir();
    }

    public void afficher() {
        System.out.println("Nom : " + nom);
        System.out.println("Téléphone : " + tel);
        System.out.println("Adresse : ");
        adresse.afficher();
    }

    // Getters / Setters
    public String getNom() { return nom; }
    public void setNom(String nom) { this.nom = nom; }

    public String getTel() { return tel; }
    public void setTel(String tel) { this.tel = tel; }

    public Adresse getAdresse() { return adresse; }
    public void setAdresse(Adresse adresse) { this.adresse = adresse; }
}

public abstract class Abonnement {

    protected Abonne client; 
    protected Date dateDebut;
    protected Date dateFin; 
    private Date dateDernierPaiement;

    protected Connexion connexion;

    public Abonnement() {}

    public void saisir() {
        System.out.println("===== Saisir Abonnement =====");

        System.out.println("Saisir date de début : ");
        dateDebut = new Date();
        dateDebut.saisir();

        System.out.println("Saisir date du dernier paiement : ");
        dateDernierPaiement = new Date();
        dateDernierPaiement.saisir();

        connexion = new Connexion();
        connexion.saisir();
    }

    public void afficher() {
        System.out.println("===== Informations Abonnement =====");
        System.out.print("Date début : ");
        dateDebut.afficher();

        if (dateFin != null) {
            System.out.print("Date fin : ");
            dateFin.afficher();
        } else {
            System.out.println("Date fin : Non résilié");
        }

        System.out.print("Dernier paiement : ");
        dateDernierPaiement.afficher();

        connexion.afficher();

        System.out.println("Client :");
        client.afficher();
    }

    public void resilier() {
        System.out.println("== Résiliation ==");
        dateFin = new Date();
        dateFin.saisir();
    }

    public boolean estResilie() {
        return dateFin != null;
    }

    // GETTERS SETTERS
    public Abonne getClient() { return client; }
    public void setClient(Abonne client) { this.client = client; }

    public Connexion getConnexion() { return connexion; }
    public void setConnexion(Connexion con) { this.connexion = con; }
}
